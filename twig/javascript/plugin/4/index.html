HTML

JS

function InputText(element, options) {
if (!element) { return false; }
var me = this;
me.container = element;
me.options = options || {};
me.options.initValue = me.options.initValue || 'Please input';
if (me.container.addEventListener) {
me.container.addEventListener('focus', function() { me.focus.call(me); }, false);
me.container.addEventListener('blur', function() { me.blur.call(me); }, false);
} else if (me.container.attachEvent) {
me.container.attachEvent('onfocus', function() { me.focus.call(me); });
me.container.attachEvent('onblur', function() { me.blur.call(me); });
}
}
InputText.prototype = {
focus: function() {
if (this.container.value == this.options.initValue) {
    this.container.value = '';
}
},
blur: function() {
if (this.container.value == '') {
    this.container.value = this.options.initValue;
}
}
};
new InputText(document.getElementById('idTxt'));
new InputText(document.getElementById('idTxt2'), {initValue: 'Please input 2'});

要点总结

  • addEventListener 兼容IE9、火狐、Chrome，attachEvent 兼容IE6、IE7；第二个参数是函数名，不能加括号，否则不等第一个参数定义的事件触发就会直接运行。
  • 使用 addEventListener/attachEvent 方法后，this指针发生变化，使用 call() 改变指针对象。
